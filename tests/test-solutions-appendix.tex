\begin{exercise}[极限]
\label{ex:appendix-limit}
计算 \(\lim_{x\to 0}\frac{\sin x}{x}\).
\end{exercise}

\begin{solution}
由基本极限可得
\[
  \lim_{x\to 0}\frac{\sin x}{x}=1.
\]
\end{solution}

\begin{exercise}[导数]
\label{ex:appendix-derivative}
计算 \(f(x)=x^2\) 的导数.
\end{exercise}

\begin{solution}
由幂函数求导公式可得 \(f'(x)=2x\).
\end{solution}
