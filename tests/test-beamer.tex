\begin{frame}{数学讲义块}
  \small
  \begin{definition}[连续]
    设函数 \(f\) 在点 \(x_0\) 的某邻域内有定义. 若
    \[
      \lim_{x \to x_0} f(x) = f(x_0),
    \]
    则称 \(f\) 在 \(x_0\) 处连续.
  \end{definition}

  \begin{keyformula}
    \[
      \lim_{x \to x_0} f(x) = A
    \]
  \end{keyformula}
\end{frame}
